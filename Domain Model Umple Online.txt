class Game {
  int numPlayers;
  int numTurns;
  1 game <@>- 2..4 Player players;
}

class Player {
  1 player --1 Profile selectedProfile;
  1 player <@>- 1..2 King selectedKing;
  1 player <@>- 1 Kingdom personalKingdom;
}

  username;
  1 player <@>- 1 StatisticSheet statSheet;
}

class StatisticSheet{
  int wins;
  int gamesPlayed;
  int losses;
  int highscore;
  int totalPoints;
  int ties;
}

  int maxHeight;

  1 <@>- * Region;

  int size;
  1 deck -<@>1 Game currentGame;

}
